事件系统

事件系统体系结构风格的说明和代码示例。

// 示例代码
                    public class Main {

                        public static void main(String[] args) {
                    
                            //创建主题
                            KWICSubject kwicSubject = new KWICSubject();
                            //创建观察者
                            Input input = new Input("C:\\Users\\14264\\Desktop\\软件体系结构\\input.txt");
                            Shift shift = new Shift(input.getLineTxt());
                            Alphabetizer alphabetizer = new Alphabetizer(shift.getKwicList());
                            Output output = new Output(alphabetizer.getKwicList(), "C:\\Users\\14264\\Desktop\\软件体系结构\\output3.txt");
                    
                            // 将观察者加入主题
                    
                            kwicSubject.addPropertyChangeListener(input);
                            kwicSubject.addPropertyChangeListener(shift);
                            kwicSubject.addPropertyChangeListener(alphabetizer);
                            kwicSubject.addPropertyChangeListener(output);
                            // 逐步调用各个观察者
                            kwicSubject.startKWIC();
                        }
                    }

管道-过滤器

管道-过滤器体系结构风格的说明和代码示例。

// 示例代码
                    public class Main {

                        public static void main(String[] args) throws IOException {
                    
                            File inFile = new File("C:\\Users\\14264\\Desktop\\软件体系结构\\input.txt");
                            File outFile = new File("C:\\Users\\14264\\Desktop\\软件体系结构\\outout4.txt");
                            Pipe pipe1 = new Pipe();
                            Pipe pipe2 = new Pipe();
                            Pipe pipe3 = new Pipe();
                            Input input = new Input(inFile, pipe1);
                            Shift shift = new Shift(pipe1, pipe2);
                            Alphabetizer alphabetizer  = new Alphabetizer(pipe2, pipe3);
                            Output output = new Output(outFile,pipe3);
                            input.transform();
                            shift.transform();
                            alphabetizer.transform();
                            output.transform();
                    
                        }
                    }
